                        //js调用
                        $('input').customInput();
                        /**
                         * 复选框、单选按钮美化
                         */
                        $.fn.customInput = function(){
                            $(this).each(function(i){
                                if($(this).is('[type=checkbox],[type=radio]')){
                                    var input = $(this);
                                    //get the associated label using the input's id
                                    var label = $('label[for='+input.attr('id')+']');
                                    //get type,for classname suffix
                                    var inputType = (input.is('[type=checkbox]')) ? 'checkbox' : 'radio';
                                    //wrap the input + label in a div
                                    $('
').insertBefore(input).append(input,label);
                                    //find all inputs in this set using the shared name attribute
                                    var allInputs = $('input[name='+input.attr('name')+']');
                                    //necessary for browsers that don't support the :hover pseudo class on labels
                                    label.hover(function(){
                                        $(this).addClass('hover');
                                        if(inputType == 'checkbox' && input.is(':checked')) {
                                            $(this).addClass('checkedHover');
                                        }
                                    },function(){
                                        $(this).removeClass('hover checkedHover');
                                    });

                                    //bind custom event, trigger it, bind click,focus,blur events
                                    input.bind('updateState',function(){
                                            if(input.is(':checked')){
                                                if(input.is(':radio')){
                                                    allInputs.each(function(){
                                                        $('label[for='+$(this).attr('id')+']').removeClass('checked');
                                                    });
                                                };
                                                label.addClass('checked');
                                            } else {
                                                label.removeClass('checked checkedHover checkedFocus');
                                            }
                                        })
                                        .trigger('updateState')
                                        .click(function(){
                                            $(this).trigger('updateState');
                                        })
                                        .focus(function(){
                                            label.addClass('focus');
                                            if(inputType == 'checkbox' && input.is(':checked')) {
                                                $(this).addClass('checkedFocus');
                                            }
                                        })
                                        .blur(function(){
                                            label.removeClass('focus checkedFocus');
                                        });
                                }
                            });
                        }
                    
                        /* wrapper divs */
                        .custom-checkbox,
                        .custom-radio {
                          position: relative;
                          float: left;
                        }
                        /*隐藏input*/
                        .custom-checkbox input,
                        .custom-radio input {
                          position: absolute;
                          left: 2px;
                          top: 3px;
                          margin: 0;
                          z-index: 0;
                          display: none;
                        }

                        .custom-checkbox label,
                        .custom-radio label {
                          display: block;
                          position: relative;
                          z-index: 1;
                          padding: 0 20px 0 25px;
                          cursor: pointer;
                        }

                        .custom-checkbox label.focus,
                        .custom-radio label.focus {
                          outline: none;
                        }

                        /* ==复选框== */
                        /*未选中状态*/
                        .custom-checkbox label {
                          background: url(../gulpdemo/img/form/bg-checkbox.png?v=74363d2709) 0 2px no-repeat;
                        }
                        /*鼠标移入状态*/
                        .custom-checkbox label.hover,
                        .custom-checkbox label.focus {
                          background-position: 0 -98px;
                        }
                        /*选中状态*/
                        .custom-checkbox label.checked {
                          background-position: 0 -48px;
                        }
                        /*选中状态下鼠标移入状态*/
                        .custom-checkbox label.checked.hover,
                        .custom-checkbox label.checked.focus {
                          background-position: 0 -147px;
                        }

                        /* ==单选按钮== */
                        /*未选中状态*/
                        .custom-radio label {
                          background: url(../gulpdemo/img/form/bg-radio.png?v=6677d9a1d2) 0 4px no-repeat;
                        }
                        /*鼠标移入状态*/
                        .custom-radio label.hover,
                        .custom-radio label.focus {
                          background-position: 0px -98px;
                        }
                        /*选中状态*/
                        .custom-radio label.checked {
                          background-position: 0 -46px;
                        }
                        /*选中状态下鼠标移入状态*/
                        .custom-radio label.checked.hover {
                          background-position: 0 -149px;
