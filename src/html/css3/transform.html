                            .circle{
                                box-sizing: border-box;
                                width:100px; height:100px;
                                border-radius:50%; background-color:blue;
                                text-align:center; color:#ffffff; line-height:100px;
                                font-family:tahoma; position:relative;
                            }
                            .border-01{
                                width:100%; height:100%;
                                border-radius:50%; border:5px solid red;
                                position:absolute; top:0; left:0;
                                z-index:1;
                                clip:rect(0 50px 100px 0px);
                                -ms-transition:all 1s ease-in-out 0.1s;
                                -moz-transition:all 1s ease-in-out 0.1s;
                                -webkit-transition:all 1s ease-in-out 0.1s;
                                -o-transition:all 1s ease-in-out 0.1s;
                                transition:all 1s ease-in-out 0.1s;
                                box-sizing:border-box;
                            }
                            .circlebg{
                                border-color: blue;
                                border-width: 6px;
                                z-index: 2;
                            }
                        
                            var circle = document.getElementById("circle");
                            var round1 = circle.querySelector(".J_round1"),
                                    round2 = circle.querySelector(".J_round2"),
                                    circlebg = circle.querySelector(".circlebg"),
                                    circleSpan = circle.querySelector('span');
                            var degree = circleSpan.innerText || circleSpan.textContent;
                            degree = Number(degree.replace('%',''));
                            if(degree<=50){
                                circlebg.style.display = 'block';

                            }else{
                                round2.style.webkitTransform = "rotate(180deg)";
                                round2.style.mozTransform = "rotate(180deg)";
                                round2.style.msTransform = "rotate(180deg)";
                                round2.style.OTransform = "rotate(180deg)";
                                round2.style.transform = "rotate(180deg)";
                                circlebg.style.display = 'none';
                            }
                            degree = 3.6*degree;
                            round1.style.webkitTransform = "rotate(" + degree +"deg)";
                            round1.style.mozTransform = "rotate(" + degree +"deg)";
                            round1.style.msTransform = "rotate(" + degree +"deg)";
                            round1.style.OTransform = "rotate(" + degree +"deg)";
                            round1.style.transform = "rotate(" + degree +"deg)";
